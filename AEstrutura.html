Estrutura de Arquivos do GNU/Linux

        /
├── bin           # Executáveis essenciais do sistema
├── boot          # Armazena arquivos necessários para a inicialização do sistema
├── dev           # Arquivos de dispositivo que representam hardware
├── etc           # Arquivos de configuração do sistema e de aplicativos
├── home          # Diretórios pessoais dos usuários
├── lib           # Bibliotecas essenciais para executáveis
├── media         # Ponto de montagem para dispositivos removíveis
├── mnt           # Ponto de montagem temporário para sistemas de arquivos
├── opt           # Pacotes de software adicionais
├── proc          # Sistema de arquivos virtual com informações sobre processos
├── root          # Diretório home do usuário root
├── run           # Informações sobre o sistema em tempo de execução
├── srv           # Dados para serviços fornecidos pelo sistema
├── sys           # Sistema de arquivos virtual com informações sobre dispositivos
├── tmp           # Diretório para arquivos temporários
├── usr           # Dados de usuário e aplicativos
│   ├── bin       # Executáveis de usuário
│   ├── lib       # Bibliotecas de aplicativos
│   └── share     # Dados compartilhados
└── var           # Arquivos variáveis, como logs e caches
